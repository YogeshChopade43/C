#include <stdio.h>
int main() {
  int n;
  double ar[100];
  printf("Enter the number of elements (1 to 100): ");
  scanf("%d", &n);

  for (int i = 0; i < n; i++) {
    printf("Enter number%d: ", i + 1);
    scanf("%d", &ar[i]);
  }

  // storing the largest number to ar[0]
  for (int i = 1; i < n; i++) {
    if (ar[0] < ar[i]) {
      ar[0] = ar[i];
    }
  }

  printf("Largest element = %d", ar[0]);

  return 0;
}
